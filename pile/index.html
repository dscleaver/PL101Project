def primes[r] = 
  (def nums[n out] = (new r1 
     ( +![n 1 r1] | 
       r1?n1 = ( out!n1 | 
                 nums![n1 out] )))
   def filter[n in out] = 
     in?m = (new r1 
       run %![m n r1] 
       new z 
       run r1?remainder = /=![remainder 0 z] 
       z?is_not_zero = if is_not_zero then out!m . filter![n in out] else filter![n in out])
   new c  
   run nums![1 c]
   new loop
   run loop!c
   loop?*in = 
     in?x = r!x.(new c1 (filter![x in c1] | loop!c1)))

new values
run primes![values]
run (values?v = log!v | values?v = log!v | values?v = log!v)
