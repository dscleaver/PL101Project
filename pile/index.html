def stream_primes[r] = 
  (def nums[n out] = (new r1 
     ( +![n 1 r1] | 
       r1?n1 = out!n1 . nums![n1 out] ))
   def filter[n in out] = 
     in?m = (new r1 
       run %![m n r1] 
       new z 
       run r1?remainder = /=![remainder 0 z] 
       z?is_not_zero = if is_not_zero then out!m . filter![n in out] else filter![n in out])
   new c new loop 
   run nums![1 c]
   run loop!c
   loop?*in = 
     in?x = r!x.(new c1 (filter![x in c1] | loop!c1)))

new primes run stream_primes![primes]

def take[n stream out] =

    run >![n 0 r]
    run -![n 1 r2]
    r?more = if more then (stream?prime = out!prime | r2?n1 = take![n1 stream out]) else ())
